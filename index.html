// Función para guardar Animal async function guardarAnimal() { const id = prompt("Caravana:"); const p = prompt("Peso (kg):"); const e = prompt("Estado:"); if(!id || !p) return; await fetch('https://agronexo-backend.onrender.com/api/ganaderia', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({caravana: id, peso: parseFloat(p), estado: e}) }); location.reload(); // Recarga para ver el cambio } // Función para guardar Lote async function guardarLote() { const n = prompt("Nombre:"); const c = prompt("Cultivo:"); const h = prompt("Has:"); if(!n) return; await fetch('https://agronexo-backend.onrender.com/api/lotes', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({nombre: n, cultivo: c, has: parseFloat(h)}) }); location.reload(); } // Función para guardar Lluvia async function guardarLluvia() { const m = prompt("Milímetros:"); if(!m) return; await fetch('https://agronexo-backend.onrender.com/api/lluvias', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({mm: parseFloat(m), fecha: new Date().toLocaleDateString()}) }); location.reload(); }
